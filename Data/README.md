[physics]
Tfinal = total simulated time in Myr
L = box size in kpc
C = Poisson's constant. i.e., 
	G*4*np.pi where G is Newton's constant, in kpc, Myr, M_solar 
D = number of spatial dimensions
m22 = mass of ultralight field field in m22 x 10^{-22} eV 

[simulation]
N = box size resolution
n_particles = number of corupuscular particles
mp_mean = mean mass of corpuscular particles
drops = number of data drops
padded = is the box isolated
c_f = courant factor
